cmake_minimum_required(VERSION 3.21)
set(CMAKE_VERBOSE_MAKEFILE ON)
project(WebScrapperGenshinCharacterCard LANGUAGES CXX)
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
add_executable(WebScrapperGenshinCharacterCard WebScrapperGenshinCharacterCard.cpp)
find_package(cpr CONFIG REQUIRED)
target_link_libraries(WebScrapperGenshinCharacterCard PRIVATE cpr::cpr)
find_package(CURL CONFIG REQUIRED)
target_link_libraries(WebScrapperGenshinCharacterCard PRIVATE CURL::libcurl)
find_package(unofficial-gumbo CONFIG REQUIRED)
target_link_libraries(WebScrapperGenshinCharacterCard PRIVATE unofficial::gumbo::gumbo)
